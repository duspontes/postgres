UPDATE LANCAMENTO SET CODPARCEIRO =70 WHERE CODPARCEIRO IN (
SELECT DISTINCT CODCLIENTE FROM CLIENTE WHERE SUBSTR(NOME, 1,3) ~ '^[-0-9]+$'  
) AND PAGREC='R' AND TIPOPARCEIRO='C'


UPDATE CUPOM SET CODCLIENTE =70 WHERE CODCLIENTE IN (
SELECT DISTINCT CODCLIENTE FROM CLIENTE WHERE SUBSTR(NOME, 1,3) ~ '^[-0-9]+$'  
) 


DELETE FROM CLIENTE WHERE SUBSTR(NOME, 1,3) ~ '^[-0-9]+$'
