WHERE LENGTH(cpfcnpj) = 11 AND tppessoa = 'F';

SUBSTR(cpfcnpj,13,2) 
 WHERE LENGTH(cpfcnpj) = 19 AND tppessoa = 'J' AND cpfcnpj IS NOT NULL
